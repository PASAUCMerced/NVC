858 // Given a trace this function adds instruction to each instruction in the trace.
859 // It also adds the trace to a hash table "GLOBAL_STATE.traceShadowMap" to maintain the reverse mapping from an (interesting) instruction's position in CCT back to its IP.
860  static inline VOID PopulateIPReverseMapAndAccountTraceInstructions(TRACE trace, uint32_t traceKey, uint32_t numInterestingInstInTrace, IsInterestingInsFptr isInterestingIns) {
861  // if there were 0 numInterestingInstInTrace, then let us simply return since it makes no sense to record anything about it.
862  if(numInterestingInstInTrace == 0)
863  return;
864 
865  ADDRINT* ipShadow = (ADDRINT*)malloc((2 + numInterestingInstInTrace) * sizeof(ADDRINT)); // +1 to hold the number of slots as a metadata and ++1 to hold module id
866  // Record the number of instructions in the trace as the first entry
867  ipShadow[0] = numInterestingInstInTrace;
868  // Record the module id as 2nd entry
869  ipShadow[1] = IMG_Id(IMG_FindByAddress(TRACE_Address(trace)));
870  uint32_t slot = 0;
871  GLOBAL_STATE.traceShadowMap[traceKey] = &ipShadow[2] ; // 0th entry is 2 behind
872 
873  for(BBL bbl = TRACE_BblHead(trace); BBL_Valid(bbl); bbl = BBL_Next(bbl)) {
874  for(INS ins = BBL_InsHead(bbl); INS_Valid(ins); ins = INS_Next(ins)) {
875  // If it is a call/ret instruction, we need to adjust the CCT.
876  // manage context
877  if(INS_IsProcedureCall(ins)) {
878  // INS_InsertPredicatedCall if the call is not made, we should not set the flag
879  INS_InsertPredicatedCall(ins, IPOINT_BEFORE, (AFUNPTR) SetCallInitFlag, IARG_UINT32, slot, IARG_THREAD_ID, IARG_END);
880 
881  if(GLOBAL_STATE.userInstrumentationCallback) {
882  // Call user instrumentation passing the flag
883  if(isInterestingIns(ins))
884  GLOBAL_STATE.userInstrumentationCallback(ins, GLOBAL_STATE.userInstrumentationCallbackArg, slot);
885  } else {
886  // TLS will remember your slot no.
887  // TODO: should this be INS_InsertPredicatedCall? not sure. One can argue either ways.
888  INS_InsertCall(ins, IPOINT_BEFORE, (AFUNPTR) RememberSlotNoInTLS, IARG_UINT32, slot, IARG_THREAD_ID, IARG_END);
889  }
890 
891  // put next slot in corresponding ins start location;
892  ipShadow[slot + 2] = INS_Address(ins); // +2 because the first 2 entries hold metadata
893  slot++;
894  } else if(INS_IsRet(ins)) {
895  // INS_InsertPredicatedCall if the RET is not made, we should not change CCT node
896  // CALL_ORDER_LAST because we want update context after all analysis routines for RET have executed.
897  INS_InsertPredicatedCall(ins, IPOINT_BEFORE, (AFUNPTR) GoUpCallChain, IARG_CALL_ORDER, CALL_ORDER_LAST, IARG_THREAD_ID, IARG_END);

905  // TODO: should this be INS_InsertPredicatedCall? not sure. One can argue either ways.
906  INS_InsertCall(ins, IPOINT_BEFORE, (AFUNPTR) RememberSlotNoInTLS, IARG_UINT32, slot, IARG_THREAD_ID, IARG_END);
907  }
908 
909  // put next slot in corresponding ins start location;
910  ipShadow[slot + 2] = INS_Address(ins); // +2 because the first 2 entries hold metadata
911  slot++;
912  } else if(isInterestingIns(ins)) {
913  if(GLOBAL_STATE.userInstrumentationCallback) {
914  // Call user instrumentation passing the flag
915  GLOBAL_STATE.userInstrumentationCallback(ins, GLOBAL_STATE.userInstrumentationCallbackArg, slot);
