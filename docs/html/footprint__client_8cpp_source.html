362 static VOID InstrumentInsCallback(INS ins, VOID* v, const uint32_t opaqueHandle) {
363  if (!INS_IsMemoryRead(ins) && !INS_IsMemoryWrite(ins)) return;
364  if (INS_IsStackRead(ins) || INS_IsStackWrite(ins)) return;
365  if (INS_IsBranchOrCall(ins) || INS_IsRet(ins)) return;
366  UINT32 memOperands = INS_MemoryOperandCount(ins);
367 
368  for(UINT32 memOp = 0; memOp < memOperands; memOp++) {
369  UINT32 refSize = INS_MemoryOperandSize(ins, memOp);
370  switch(refSize) {
371 
372  HANDLE_CASE(1);
373  HANDLE_CASE(2);
374  HANDLE_CASE(4);
375  HANDLE_CASE(8);
376  HANDLE_CASE(10); // TODO
377  HANDLE_CASE(16); // TODO
378 
379  default: {
380  // seeing some stupid 512 (fxsave)byte operations. Suspecting REP-instructions.
381  if(INS_MemoryOperandIsRead(ins, memOp)) {
382  INS_InsertPredicatedCall(ins, IPOINT_BEFORE, (AFUNPTR) RecordLargeMemRead, IARG_MEMORYOP_EA, memOp, IARG_MEMORYREAD_SIZE, IARG_UINT32, opaqueHandle, IARG_INST_PTR, IARG_THREAD_ID, IARG_END);
383  }
384 
385  if(INS_MemoryOperandIsWritten(ins, memOp)) {
386  INS_InsertPredicatedCall(ins, IPOINT_BEFORE, (AFUNPTR) RecordLargeMemWrite, IARG_MEMORYOP_EA, memOp, IARG_MEMORYWRITE_SIZE,
387  IARG_UINT32, opaqueHandle, IARG_INST_PTR,
388  IARG_THREAD_ID, IARG_END);
389  }
390  }
391  break;
392  }
393  }
