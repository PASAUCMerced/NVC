58 void ClientInit(int argc, char* argv[]) {
59  // Create output file
60  char name[MAX_FILE_PATH] = "client.out.";
61  char* envPath = getenv("CCTLIB_CLIENT_OUTPUT_FILE");
62 
63  if(envPath) {
64  // assumes max of MAX_FILE_PATH
65  strcpy(name, envPath);
66  }
67 
68  gethostname(name + strlen(name), MAX_FILE_PATH - strlen(name));
69  pid_t pid = getpid();
70  sprintf(name + strlen(name), "%d", pid);
71  cerr << "\n Creating log file at:" << name << "\n";
72  gTraceFile = fopen(name, "w");
73  // print the arguments passed
74  fprintf(gTraceFile, "\n");
75 
76  for(int i = 0 ; i < argc; i++) {
77  fprintf(gTraceFile, "%s ", argv[i]);
78  }
79 
80  fprintf(gTraceFile, "\n");
81 }
