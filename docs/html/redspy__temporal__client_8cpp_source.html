383 static inline uint32_t GetAliasIDs(REG reg){
384  uint8_t regGroup = 0;
385  uint8_t byteInd = 0;
386  uint8_t type = 0;
387  switch (reg) {
388  case REG_RAX: regGroup = ALIAS_REG_A; byteInd = ALIAS_BYTES_INDEX_64; type = ALIAS_GENERIC; break;
389  case REG_EAX: regGroup = ALIAS_REG_A; byteInd = ALIAS_BYTES_INDEX_32; type = ALIAS_GENERIC; break;
390  case REG_AX: regGroup = ALIAS_REG_A; byteInd = ALIAS_BYTES_INDEX_16; type = ALIAS_GENERIC; break;
391  case REG_AH: regGroup = ALIAS_REG_A; byteInd = ALIAS_BYTES_INDEX_8_H; type = ALIAS_HIGH_BYTE; break;
392  case REG_AL: regGroup = ALIAS_REG_A; byteInd = ALIAS_BYTES_INDEX_8_L; type = ALIAS_LOW_BYTE; break;
393 
394  case REG_RBX: regGroup = ALIAS_REG_B; byteInd = ALIAS_BYTES_INDEX_64; type = ALIAS_GENERIC; break;
395  case REG_EBX: regGroup = ALIAS_REG_B; byteInd = ALIAS_BYTES_INDEX_32; type = ALIAS_GENERIC; break;
396  case REG_BX: regGroup = ALIAS_REG_B; byteInd = ALIAS_BYTES_INDEX_16; type = ALIAS_GENERIC; break;
397  case REG_BH: regGroup = ALIAS_REG_B; byteInd = ALIAS_BYTES_INDEX_8_H; type = ALIAS_HIGH_BYTE; break;
398  case REG_BL: regGroup = ALIAS_REG_B; byteInd = ALIAS_BYTES_INDEX_8_L; type = ALIAS_LOW_BYTE; break;
399 
400  case REG_RCX: regGroup = ALIAS_REG_C; byteInd = ALIAS_BYTES_INDEX_64; type = ALIAS_GENERIC; break;
401  case REG_ECX: regGroup = ALIAS_REG_C; byteInd = ALIAS_BYTES_INDEX_32; type = ALIAS_GENERIC; break;
402  case REG_CX: regGroup = ALIAS_REG_C; byteInd = ALIAS_BYTES_INDEX_16; type = ALIAS_GENERIC; break;
403  case REG_CH: regGroup = ALIAS_REG_C; byteInd = ALIAS_BYTES_INDEX_8_H; type = ALIAS_HIGH_BYTE; break;
404  case REG_CL: regGroup = ALIAS_REG_C; byteInd = ALIAS_BYTES_INDEX_8_L; type = ALIAS_LOW_BYTE; break;
405 
406  case REG_RDX: regGroup = ALIAS_REG_D; byteInd = ALIAS_BYTES_INDEX_64; type = ALIAS_GENERIC; break;
407  case REG_EDX: regGroup = ALIAS_REG_D; byteInd = ALIAS_BYTES_INDEX_32; type = ALIAS_GENERIC; break;
408  case REG_DX: regGroup = ALIAS_REG_D; byteInd = ALIAS_BYTES_INDEX_16; type = ALIAS_GENERIC; break;
409  case REG_DH: regGroup = ALIAS_REG_D; byteInd = ALIAS_BYTES_INDEX_8_H; type = ALIAS_HIGH_BYTE; break;
410  case REG_DL: regGroup = ALIAS_REG_D; byteInd = ALIAS_BYTES_INDEX_8_L; type = ALIAS_LOW_BYTE; break;
411  default: assert(0 && "not alias registers! should not reach here!"); break;
412  }
413  uint32_t aliasGroupByteType = ((uint32_t)regGroup << 16) | ((uint32_t)byteInd << 8) | ((uint32_t)type);
414  return aliasGroupByteType;
415 }

478 static inline void InstrumentAliasReg(INS ins, REG reg, uint32_t opaqueHandle){
479 
480  uint32_t aliasIDs = GetAliasIDs(reg);
481  uint8_t regId = static_cast<uint8_t>(((aliasIDs) & 0x00ffffff) >> 16 );
482  uint8_t byteOffset = static_cast<uint8_t>(((aliasIDs) & 0x0000ffff) >> 8 );
483  uint8_t type;
484

512 template<int start, int end, int incr, bool conditional>
513 struct UnrolledLoop{
514  static __attribute__((always_inline)) void Body(function<void (const int)> func){
515  func(start); // Real loop body
516  UnrolledLoop<start+incr, end, incr, conditional>:: Body(func); // unroll next iteration
517  }
518  static __attribute__((always_inline)) void BodySamePage(ContextHandle_t * __restrict__ prevIP, const ContextHandle_t handle, THREADID threadId){
519  if(conditional) {
520  // report in RedTable
521  AddToRedTable(MAKE_CONTEXT_PAIR(prevIP[start], handle), 1, threadId);
522  }
523  // Update context
524  prevIP[start] = handle;
525  UnrolledLoop<start+incr, end, incr, conditional>:: BodySamePage(prevIP, handle, threadId); // unroll next iteration
526  }
527  static __attribute__((always_inline)) void BodyStraddlePage(uint64_t addr, const ContextHandle_t handle, THREADID threadId){
528  uint8_t * status = GetOrCreateShadowBaseAddress((uint64_t)addr + start);
529  ContextHandle_t * prevIP = (ContextHandle_t*)(status + PAGE_OFFSET(((uint64_t)addr + start)) * sizeof(ContextHandle_t));
530  if (conditional) {

532  AddToRedTable(MAKE_CONTEXT_PAIR(prevIP[0 /* 0 is correct*/ ], handle), 1, threadId);
533  }
534  // Update context
535  prevIP[0] = handle;
536  UnrolledLoop<start+incr, end, incr, conditional>:: BodyStraddlePage(addr, handle, threadId); // unroll next iteration
537  }
538 };

599  static __attribute__((always_inline)) VOID CheckNByteValueAfterWrite(uint32_t opaqueHandle, THREADID threadId){
600  RedSpyThreadData* const tData = ClientGetTLS(threadId);
601  void * addr;
602  bool isRedundantWrite = IsWriteRedundant(addr, threadId);
603 
604  ContextHandle_t curCtxtHandle = GetContextHandle(threadId, opaqueHandle);
605 
606  uint8_t* status = GetOrCreateShadowBaseAddress((uint64_t)addr);
607  ContextHandle_t * __restrict__ prevIP = (ContextHandle_t*)(status + PAGE_OFFSET((uint64_t)addr) * sizeof(ContextHandle_t));
608  const bool isAccessWithinPageBoundary = IS_ACCESS_WITHIN_PAGE_BOUNDARY( (uint64_t)addr, AccessLen);
609  if(isRedundantWrite) {
610  // detected redundancy
611  if(isAccessWithinPageBoundary) {
612  // All from same ctxt?
613  if (UnrolledConjunction<0, AccessLen, 1>::BodyContextCheck(prevIP)) {
614  // report in RedTable
615  AddToRedTable(MAKE_CONTEXT_PAIR(prevIP[0], curCtxtHandle), AccessLen, threadId);
616  // Update context
617  UnrolledLoop<0, AccessLen, 1, false /* redundancy is updated outside*/>::BodySamePage(prevIP, curCtxtHandle, threadId);
618  } else {
619  // different contexts
620  UnrolledLoop<0, AccessLen, 1, true /* redundancy is updated inside*/>::BodySamePage(prevIP, curCtxtHandle, threadId);
621  }
622  } else {
623  // Write across a 64-K page boundary
624  // First byte is on this page though
625  AddToRedTable(MAKE_CONTEXT_PAIR(prevIP[0], curCtxtHandle), 1, threadId);
626  // Update context
627  prevIP[0] = curCtxtHandle;
628 
629  // Remaining bytes [1..AccessLen] somewhere will across a 64-K page boundary
630  UnrolledLoop<1, AccessLen, 1, true /* update redundancy */>::BodyStraddlePage( (uint64_t) addr, curCtxtHandle, threadId);
631  }
632  } else {
633  // No redundancy.
634  // Just update contexts
635  if(isAccessWithinPageBoundary) {
636  // Update context
637  UnrolledLoop<0, AccessLen, 1, false /* not redundant*/>::BodySamePage(prevIP, curCtxtHandle, threadId);
638  } else {
639  // Write across a 64-K page boundary
640  // Update context
641  prevIP[0] = curCtxtHandle;
642 
643  // Remaining bytes [1..AccessLen] somewhere will across a 64-K page boundary
644  UnrolledLoop<1, AccessLen, 1, false /* dont update redundancy */>::BodyStraddlePage( (uint64_t) addr, curCtxtHandle, threadId);
645 
646  }
647  }
648  }

774 static VOID InstrumentInsCallback(INS ins, VOID* v, const uint32_t opaqueHandle) {
775  if (!INS_HasFallThrough(ins)) return;
776  if (INS_IsIgnorable(ins))return;
777  if (INS_IsBranchOrCall(ins) || INS_IsRet(ins)) return;
778 
779  //Instrument memory writes to find redundancy
780  // Special case, if we have only one write operand
781  UINT32 whichOp = 0;
782  if(GetNumWriteOperandsInIns(ins, whichOp) == 1){
783  // Read the value at location before and after the instruction
784  RedSpyInstrument<0>::InstrumentReadValueBeforeAndAfterWriting(ins, whichOp, opaqueHandle);
785  }else{
786  UINT32 memOperands = INS_MemoryOperandCount(ins);
787  int readBufferSlotIndex=0;
788  for(UINT32 memOp = 0; memOp < memOperands; memOp++) {
789 
790  if(!INS_MemoryOperandIsWritten(ins, memOp))
791  continue;
792 
793  switch (readBufferSlotIndex) {
794  case 0:
795  // Read the value at location before and after the instruction
796  RedSpyInstrument<0>::InstrumentReadValueBeforeAndAfterWriting(ins, whichOp, opaqueHandle);
797  break;
798  case 1:
799  // Read the value at location before and after the instruction
800  RedSpyInstrument<1>::InstrumentReadValueBeforeAndAfterWriting(ins, whichOp, opaqueHandle);
801  break;
802  case 2:
803  // Read the value at location before and after the instruction
804  RedSpyInstrument<2>::InstrumentReadValueBeforeAndAfterWriting(ins, whichOp, opaqueHandle);
805  break;
806  case 3:
807  // Read the value at location before and after the instruction
808  RedSpyInstrument<3>::InstrumentReadValueBeforeAndAfterWriting(ins, whichOp, opaqueHandle);
809  break;
810  case 4:
811  // Read the value at location before and after the instruction
812  RedSpyInstrument<4>::InstrumentReadValueBeforeAndAfterWriting(ins, whichOp, opaqueHandle);
813  break;
814  default:
815  assert(0 && "NYI");
816  break;
817  }
818  // use next slot for the next write operand
819  readBufferSlotIndex++;
820  }
821  }
822 
823  //Instrument register writes to find redundancy
824  UINT32 numOperands = INS_OperandCount(ins);
825 
826  for(UINT32 oper = 0; oper < numOperands; oper++) {
827 
828  if(!INS_OperandWritten(ins, oper) || !INS_OperandIsReg(ins,oper))
829  continue;
830 
831  REG reg = INS_OperandReg(ins,oper);
832 
833  if(REG_IsIgnorable(reg))
834  continue;
835 
836  if (RegHasAlias(reg)) {
837  switch (reg) {
838  case REG_RAX: HANDLE_ALIAS_GENERIC(uint64_t, ALIAS_REG_A, ALIAS_BYTES_INDEX_64); break;
839  case REG_EAX: HANDLE_ALIAS_GENERIC(uint32_t, ALIAS_REG_A, ALIAS_BYTES_INDEX_32); break;
840  case REG_EBX: HANDLE_ALIAS_GENERIC(uint32_t, ALIAS_REG_B, ALIAS_BYTES_INDEX_32); break;
841  case REG_ECX: HANDLE_ALIAS_GENERIC(uint32_t, ALIAS_REG_C, ALIAS_BYTES_INDEX_32); break;
842  case REG_EDX: HANDLE_ALIAS_GENERIC(uint32_t, ALIAS_REG_D, ALIAS_BYTES_INDEX_32); break;
843  default: InstrumentAliasReg(ins, reg , opaqueHandle); break;
844  }
845  }else{
846  InstrumentGeneralReg(ins, reg, opaqueHandle);
847  }
848  }
