244 struct UnrolledLoop{
245  static __attribute__((always_inline)) void Body(function<void (const int)> func){
246  func(start); // Real loop body
247  UnrolledLoop<start+incr, end, incr, conditional>:: Body(func); // unroll next iteration
248  }
249  static __attribute__((always_inline)) void BodySamePage(ContextHandle_t * __restrict__ prevIP, const ContextHandle_t handle, THREADID threadId){
250  if(conditional) {
251  // report in RedTable
252  AddToRedTable(MAKE_CONTEXT_PAIR(prevIP[start], handle), 1, threadId);
253  }
254  // Update context
255  prevIP[start] = handle;
256  UnrolledLoop<start+incr, end, incr, conditional>:: BodySamePage(prevIP, handle, threadId); // unroll next iteration
257  }
258  static __attribute__((always_inline)) void BodyStraddlePage(uint64_t addr, const ContextHandle_t handle, THREADID threadId){
259  uint8_t * status = GetOrCreateShadowBaseAddress((uint64_t)addr + start);
260  ContextHandle_t * prevIP = (ContextHandle_t*)(status + PAGE_OFFSET(((uint64_t)addr + start)) * sizeof(ContextHandle_t));
261  if (conditional) {
262  // report in RedTable
263  AddToRedTable(MAKE_CONTEXT_PAIR(prevIP[0 /* 0 is correct*/ ], handle), 1, threadId);
264  }
265  // Update context
266  prevIP[0] = handle;
267  UnrolledLoop<start+incr, end, incr, conditional>:: BodyStraddlePage(addr, handle, threadId); // unroll next iteration
268  }

299 struct RedSpyAnalysis{
300  static inline bool IsWriteRedundant(void * &addr, THREADID threadId){
301  RedSpyThreadData* const tData = ClientGetTLS(threadId);
302  AddrValPair * avPair = & tData->buffer[bufferOffset];
303  addr = avPair->address;
304  if(AccessLen >= 4){
305  double newvalue64 = *(static_cast<double*>(avPair->address));
306  double oldvalue64 = *((double*)(&avPair->value));
307  double rate64 = (newvalue64 - oldvalue64)/oldvalue64;
308  *((double*)(&avPair->value)) = newvalue64;
309  if( rate64 <= delta || rate64 >= -delta )
310  return true;
311  else
312  return false;
313  }else{
314  return 0;
315  }
316  }

332  static __attribute__((always_inline)) VOID CheckNByteValueAfterWrite(uint32_t opaqueHandle, THREADID threadId){
333  RedSpyThreadData* const tData = ClientGetTLS(threadId);
334  void * addr;
335  bool isRedundantWrite = IsWriteRedundant(addr, threadId);
336 
337  ContextHandle_t curCtxtHandle = GetContextHandle(threadId, opaqueHandle);
338 
339  uint8_t* status = GetOrCreateShadowBaseAddress((uint64_t)addr);
340  ContextHandle_t * __restrict__ prevIP = (ContextHandle_t*)(status + PAGE_OFFSET((uint64_t)addr) * sizeof(ContextHandle_t));
341  const bool isAccessWithinPageBoundary = IS_ACCESS_WITHIN_PAGE_BOUNDARY( (uint64_t)addr, AccessLen);
342  if(isRedundantWrite) {
343  // detected redundancy
344  if(isAccessWithinPageBoundary) {
345  // All from same ctxt?
346  if (UnrolledConjunction<0, AccessLen, 1>::BodyContextCheck(prevIP)) {
347  // report in RedTable
348  AddToRedTable(MAKE_CONTEXT_PAIR(prevIP[0], curCtxtHandle), AccessLen, threadId);
349  // Update context
350  UnrolledLoop<0, AccessLen, 1, false /* redundancy is updated outside*/>::BodySamePage(prevIP, curCtxtHandle, threadId);
351  } else {
352  // different contexts
353  UnrolledLoop<0, AccessLen, 1, true /* redundancy is updated inside*/>::BodySamePage(prevIP, curCtxtHandle, threadId);
354  }
355  } else {
356  // Write across a 64-K page boundary
357  // First byte is on this page though
358  AddToRedTable(MAKE_CONTEXT_PAIR(prevIP[0], curCtxtHandle), 1, threadId);
359  // Update context
360  prevIP[0] = curCtxtHandle;
361 
362  // Remaining bytes [1..AccessLen] somewhere will across a 64-K page boundary
363  UnrolledLoop<1, AccessLen, 1, true /* update redundancy */>::BodyStraddlePage( (uint64_t) addr, curCtxtHandle, threadId);
364  }
365  } else {
366  // No redundancy.
367  // Just update contexts
368  if(isAccessWithinPageBoundary) {
369  // Update context
370  UnrolledLoop<0, AccessLen, 1, false /* not redundant*/>::BodySamePage(prevIP, curCtxtHandle, threadId);
371  } else {
372  // Write across a 64-K page boundary
373  // Update context
374  prevIP[0] = curCtxtHandle;
375 
376  // Remaining bytes [1..AccessLen] somewhere will across a 64-K page boundary
377  UnrolledLoop<1, AccessLen, 1, false /* dont update redundancy */>::BodyStraddlePage( (uint64_t) addr, curCtxtHandle, threadId);
378 
379  }
380  }
381  }

450 static VOID InstrumentInsCallback(INS ins, VOID* v, const uint32_t opaqueHandle) {
451  if (!INS_IsMemoryRead(ins) && !INS_IsMemoryWrite(ins)) return;
452  // if (INS_IsStackRead(ins) || INS_IsStackWrite(ins)) return;
453  if (INS_IsBranchOrCall(ins) || INS_IsRet(ins)) return;
454 
455  if(!IsFloatInstruction(INS_Address(ins)))
456  return;
457 
458  // Special case, if we have only one write operand
459  UINT32 whichOp = 0;
460  if(GetNumWriteOperandsInIns(ins, whichOp) == 1){
461  // Read the value at location before and after the instruction
462  RedSpyInstrument<0>::InstrumentReadValueBeforeAndAfterWriting(ins, whichOp, opaqueHandle);
463  return;
464  }
465 
466  UINT32 memOperands = INS_MemoryOperandCount(ins);
467  int readBufferSlotIndex=0;
468  for(UINT32 memOp = 0; memOp < memOperands; memOp++) {
469 
470  if(!INS_MemoryOperandIsWritten(ins, memOp))
471  continue;
472 
473  switch (readBufferSlotIndex) {
474  case 0:
475  // Read the value at location before and after the instruction
476  RedSpyInstrument<0>::InstrumentReadValueBeforeAndAfterWriting(ins, whichOp, opaqueHandle);
477  break;
478  case 1:
479  // Read the value at location before and after the instruction
480  RedSpyInstrument<1>::InstrumentReadValueBeforeAndAfterWriting(ins, whichOp, opaqueHandle);
481  break;
482  case 2:
483  // Read the value at location before and after the instruction
484  RedSpyInstrument<2>::InstrumentReadValueBeforeAndAfterWriting(ins, whichOp, opaqueHandle);
485  break;
486  case 3:
487  // Read the value at location before and after the instruction
488  RedSpyInstrument<3>::InstrumentReadValueBeforeAndAfterWriting(ins, whichOp, opaqueHandle);
489  break;
490  case 4:
491  // Read the value at location before and after the instruction
492  RedSpyInstrument<4>::InstrumentReadValueBeforeAndAfterWriting(ins, whichOp, opaqueHandle);
493  break;
494  default:
495  assert(0 && "NYI");
496  break;
497  }
498 
499  // use next slot for the next write operand
500  readBufferSlotIndex++;
501  }

604 static void PrintRedundancyPairs(THREADID threadId) {
605  vector<RedundacyData> tmpList;
606  vector<RedundacyData>::iterator tmpIt;
607 
608  uint64_t grandTotalRedundantBytes = 0;
609  fprintf(gTraceFile, "*************** Dump Data(delta=%.2f%%) from Thread %d ****************\n", delta*100,threadId);
610 
611 #ifdef MERGING
612  for (unordered_map<uint64_t, uint64_t>::iterator it = RedMap[threadId].begin(); it != RedMap[threadId].end(); ++it) {
613  ContextHandle_t dead = DECODE_DEAD((*it).first);
614  ContextHandle_t kill = DECODE_KILL((*it).first);
615 
616  for(tmpIt = tmpList.begin();tmpIt != tmpList.end(); ++tmpIt){
617  if(dead == 0 || ((*tmpIt).dead) == 0){
618  continue;
619  }
620  if (!HaveSameCallerPrefix(dead,(*tmpIt).dead)) {
621  continue;
622  }
623  if (!HaveSameCallerPrefix(kill,(*tmpIt).kill)) {
624  continue;
625  }
626  bool ct1 = IsSameSourceLine(dead,(*tmpIt).dead);
627  bool ct2 = IsSameSourceLine(kill,(*tmpIt).kill);
628  if(ct1 && ct2){
629  (*tmpIt).frequency += (*it).second;
630  grandTotalRedundantBytes += (*it).second;
631  break;
632  }
633  }
634  if(tmpIt == tmpList.end()){
635  RedundacyData tmp = { dead, kill, (*it).second};
636  tmpList.push_back(tmp);
637  grandTotalRedundantBytes += tmp.frequency;
638  }
639  }
640 #else
641  for (unordered_map<uint64_t, uint64_t>::iterator it = RedMap[threadId].begin(); it != RedMap[threadId].end(); ++it) {
642  RedundacyData tmp = { DECODE_DEAD ((*it).first), DECODE_KILL((*it).first), (*it).second};
643  tmpList.push_back(tmp);
644  grandTotalRedundantBytes += tmp.frequency;
645  }
646 #endif
647 
648  fprintf(gTraceFile, "\n Total redundant bytes = %f %%\n", grandTotalRedundantBytes * 100.0 / ClientGetTLS(threadId)->bytesWritten);
649 
650  sort(tmpList.begin(), tmpList.end(), RedundacyCompare);
651  vector<struct AnalyzedMetric_t>::iterator listIt;
652  int cntxtNum = 0;
653  for (vector<RedundacyData>::iterator listIt = tmpList.begin(); listIt != tmpList.end(); ++listIt) {
654  if (cntxtNum < MAX_REDUNDANT_CONTEXTS_TO_LOG) {
655  fprintf(gTraceFile, "\n======= (%f) %% ======\n", (*listIt).frequency * 100.0 / grandTotalRedundantBytes);
656  if ((*listIt).dead == 0) {
657  fprintf(gTraceFile, "\n Prepopulated with by OS\n");
658  } else {
659  PrintFullCallingContext((*listIt).dead);
660  }
661  fprintf(gTraceFile, "\n---------------------Redundantly written by---------------------------\n");
662  PrintFullCallingContext((*listIt).kill);
663  }
664  else {
665  break;
666  }
667  cntxtNum++;
668  }
