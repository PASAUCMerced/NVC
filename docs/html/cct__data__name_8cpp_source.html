397 VOID InstrumentInsCallback(INS ins, VOID* v, const uint32_t slot) {
398  INS_InsertPredicatedCall(ins, IPOINT_BEFORE, (AFUNPTR)SimpleCCTQuery, IARG_THREAD_ID, IARG_UINT32, slot, IARG_END);
399 
400  // Data centric for mem inst
401  // Skip call, ret and JMP instructions
402  if(INS_IsBranchOrCall(ins) || INS_IsRet(ins)) {
403  return;
404  }
405 
406  // skip stack ... actually our code handles it
407  if(INS_IsStackRead(ins) || INS_IsStackWrite(ins))
408  return;
409 
410  if(INS_IsMemoryRead(ins) || INS_IsMemoryWrite(ins)) {
411  // How may memory operations?
412  UINT32 memOperands = INS_MemoryOperandCount(ins);
413 
414  // Iterate over each memory operand of the instruction and add Analysis routine
415  for(UINT32 memOp = 0; memOp < memOperands; memOp++) {
416  INS_InsertPredicatedCall(ins, IPOINT_BEFORE, (AFUNPTR) MemAnalysisRoutine, IARG_MEMORYOP_EA, memOp, IARG_THREAD_ID, IARG_END);
417  }
418  }
419 }
